variable names:
variable and data members: underscores, exp: file_name, mean_field
data members: prependend with _, exp: _t, _x
functions, members and classes: CamelCase, exp: TestClass, RayleighModel
constants: k + CamelCase, exp: kNSteps, kOscillators

spacing:
indentation: 2 spaces; no indentation after: namespace, template
vertical spacing: 1 line, where it helps readability, 2 lines between
  functions, classes and members

comments:
normal comments: //
docstrings: /* */

misc:
list-initialization in new line
include: std, project, others
